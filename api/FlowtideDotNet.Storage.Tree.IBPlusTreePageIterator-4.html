A view over a single leaf page yielded while iterating a B+ tree. It enumerates the page's key/value entries, exposes the page's Keys and Values containers, supports write locking, and can persist in-place edits through SavePage(bool).

public interface IBPlusTreePageIterator<K, V, TKeyContainer, TValueContainer> : IEnumerable<KeyValuePair<K, V>>, IEnumerable, ILockableObject where TKeyContainer : IKeyContainer<K> where TValueContainer : IValueContainer<V>
