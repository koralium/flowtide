definition user {}

definition role {
  relation can_view: user:*
}

definition role_binding {
  relation user: user
  relation role: role

  permission can_view = user & role->can_view 
}

definition organization {
  relation role_binding: role_binding
  relation parent: organization

  permission can_view = role_binding->can_view + parent->can_view
}

definition project {
  relation role_binding: role_binding
  relation organization: organization
  relation all_can_read: user:*

  permission can_view = role_binding->can_view + organization->can_view
  permission can_view_wildcard = all_can_read + role_binding->can_view + organization->can_view
}
